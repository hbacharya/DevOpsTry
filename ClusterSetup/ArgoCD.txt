# Create the namespace for ArgoCD
kubectl create namespace argocd

# Apply the official installation manifest
kubectl apply -n argocd -f https://raw.githubusercontent.com/argoproj/argo-cd/stable/manifests/install.yaml

# ArgoCD Networking
#	By default, ArgoCD installs as a ClusterIP (internal only). We want to change it to a LoadBalancer so MetalLB gives it an IP from your 192.168.1.x range.

# Patch the service to be a LoadBalancer
kubectl patch svc argocd-server -n argocd -p '{"spec": {"type": "LoadBalancer"}}'


# Check for the IP
# 	watch  services. It might take a minute for the pods to pull and start up.
# 	Look for argocd-server. You should see an EXTERNAL-IP (likely 192.168.1.150 or 151).
kubectl get svc -n argocd

# Get the first time credentials
kubectl -n argocd get secret argocd-initial-admin-secret -o jsonpath="{.data.password}" | ForEach-Object { [System.Text.Encoding]::UTF8.GetString([System.Convert]::FromBase64String($_)) }

# Login using "admin" and the password provided in above command
