# Install conntrack and socat
# 	Kubernetes requires a tool called conntrack (Connection Tracking). This is a kernel feature that allows the system to keep track of all the network connections (like which Pod is talking to which Service).
# 	Without this, the Kubernetes networking rules (iptables) can't function.
# 	think of conntrack as the State Table for your network.
# 	When a request comes into your Load Balancer and needs to go to a specific Pod, conntrack remembers where that packet came from so it can send the response back to the right place. Without it, Kubernetes is "forgetful" and packets get lost.

sudo apt install -y conntrack

sudo apt install -y socat

----------------------------------------------------

# Networking "Pipes" for kubernetes 

# 	Before the container engine can talk, we have to open the Linux kernel pipes.
# 	The Kernel Modules (overlay and br_netfilter). Think of these like Drivers you install on a Windows machine.
# 	overlay: This allows Kubernetes to "stack" file systems. It lets a container have its own read/write layer on top of a read-only base image.
#	
#	br_netfilter: This is the most critical one. It allows the Linux "Bridge" (the virtual switch inside the VM) to send traffic through the "Netfilter" (the firewall/router).
#
#	Without this: Your Pods wouldn't be able to talk to the outside world because the kernel would ignore their traffic.
#
# 	Essentially, we told the OS: "You aren't just a host anymore; you are a sophisticated network router."

cat <<EOF | sudo tee /etc/modules-load.d/k8s.conf
overlay
br_netfilter
EOF

sudo modprobe overlay
sudo modprobe br_netfilter

cat <<EOF | sudo tee /etc/sysctl.d/k8s.conf
net.bridge.bridge-nf-call-iptables  = 1
net.bridge.bridge-nf-call-ip6tables = 1
net.ipv4.ip_forward                 = 1
EOF

sudo sysctl --system

# setup SystemdCgroup

# Kubernetes requires a specific setting called SystemdCgroup to be true. If you don't do this, the "Brain" and the "Engine" will fight over who controls the processes.

# Create the default config file
sudo mkdir -p /etc/containerd
containerd config default | sudo tee /etc/containerd/config.toml > /dev/null

# Tell it to use Systemd
sudo sed -i 's/SystemdCgroup = false/SystemdCgroup = true/g' /etc/containerd/config.toml

# run the restart to pick up the new config
sudo systemctl restart containerd

# Install K8s

# Adding the "Trusted" Source
# Ubuntu’s default repositories don't always have the latest Kubernetes versions. We need to tell Ubuntu to trust the official Kubernetes source.

sudo apt-get install -y apt-transport-https ca-certificates curl gpg

# Download the official Google/K8s security key
curl -fsSL https://pkgs.k8s.io/core:/stable:/v1.31/deb/Release.key | sudo gpg --dearmor -o /etc/apt/keyrings/kubernetes-apt-keyring.gpg

# Add the K8s repo to your sources list
echo 'deb [signed-by=/etc/apt/keyrings/kubernetes-apt-keyring.gpg] https://pkgs.k8s.io/core:/stable:/v1.31/deb/ /' | sudo tee /etc/apt/sources.list.d/kubernetes.list

sudo apt-get update

# Installing the Tools (kubelet kubeadm kubectl)

#		Before the Master can take charge, it needs its tools. We need to install three specific binaries that work together. 
#		As a .NET dev, think of these like the Runtime, the CLI, and the Service Host.
#		
#		kubeadm (The Installer): The tool used to bootstrap the cluster. It’s like the "Setup.exe."
#		
#		kubelet (The Agent): This is the most important service. It runs on every node and ensures that containers are actually running. If kubelet stops, the node dies.
#		
#		kubectl (The Remote Control): The command-line utility you’ll use from your Windows machine (or the Master) to talk to the cluster.


#install kubelet kubeadm kubectl
sudo apt-get install -y kubelet kubeadm kubectl

# This "pins" the version so 'apt upgrade' doesn't break your cluster later
sudo apt-mark hold kubelet kubeadm kubectl

# kubeadm init
# 	run the command that officially creates the cluster. 
# 	Because we decided on Calico for networking, we must provide a "Pod Network" range. Calico defaults to 192.168.0.0/16.
# 	Why these flags matter:
#	--pod-network-cidr: This tells Kubernetes: "All my internal Pods will live in the 192.168.0.x range." (Don't worry, this won't conflict with your physical 192.168.1.x network because it's virtual).
#	--apiserver-advertise-address: This tells the cluster: "Hey, I am the Master, and my real IP is 192.168.1.100. Everyone should talk to me here."

# kubeadm init (on control plane)
sudo kubeadm init --pod-network-cidr=192.168.0.0/16 --apiserver-advertise-address=192.168.1.100

# Before you can run any commands, you have to tell kubectl where your cluster is. When you ran kubeadm init on the Master, it created a configuration file. You need to link that to your user profile.
mkdir -p $HOME/.kube
sudo cp -i /etc/kubernetes/admin.conf $HOME/.kube/config
sudo chown $(id -u):$(id -g) $HOME/.kube/config
